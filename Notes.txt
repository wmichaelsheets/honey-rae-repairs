  emergency boolean

}

  id int pk
  address varchar
  phoneNumber varchar
  userId int
}

  id int pk
  employeeId int

}

  id int pk
  specialty varchar

  userId int
}

  id int pk
  fullName varchar
  email varchar
  isStaff boolean
}
